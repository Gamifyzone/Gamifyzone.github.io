export enum AppState {
LANDING = 'LANDING',
ADMISSIONS_FORM = 'ADMISSIONS_FORM',
ENTRANCE_EXAM = 'ENTRANCE_EXAM',
ADMISSIONS_RESULT = 'ADMISSIONS_RESULT',
DASHBOARD = 'DASHBOARD',
CLASSROOM = 'CLASSROOM',
EXAM_HALL = 'EXAM_HALL',
GRADUATION = 'GRADUATION'
}
export enum Role {
STUDENT = 'Student',
FACULTY = 'Faculty',
REGISTRAR = 'Registrar',
EXAMINER = 'Examiner',
ADVISOR = 'Academic Advisor'
}
export interface StudentProfile {
name: string;
fieldOfStudy: string;
gpa: number;
creditsEarned: number;
currentSemester: number; // 1 or 2
currentWeek: number; // 1 to 24 (approx 6 months per semester)
status: 'Applicant' | 'Enrolled' | 'Probation' | 'Graduated';
feedbackHistory: string[];
}
export interface Course {
id: string;
code: string;
title: string;
description: string;
credits: number;
completed: boolean;
grade?: number;
syllabus: string[]; // List of weekly topics
}
export interface Curriculum {
degreeTitle: string;
courses: Course[];
}
export interface QuizQuestion {
question: string;
options: string[];
correctOptionIndex: number; // For internal checking if AI provides it, otherwise we ask AI to grade
}
export interface ExamResult {
score: number;
feedback: string;
passed: boolean;
}
export interface Message {
role: 'user' | 'model' | 'system'; // system used for UI hints
senderName: string;
content: string;
timestamp: number;
}
