import React, { useState, useEffect, useRef } from 'react';
import ReactMarkdown from 'react-markdown';
import { Course, QuizQuestion } from '../types';
import { generateLecture, generateCourseQuiz, chatWithAdvisor } from '../services/geminiService';
import { ArrowLeft, BookOpen, MessageSquare, Check, Loader2, Send, GraduationCap } from 'lucide-react';
interface ClassroomProps {
course: Course;
week: number;
onBack: () => void;
onCompleteCourse: (courseId: string, grade: number) => void;
}
const Classroom: React.FC<ClassroomProps> = ({ course, week, onBack, onCompleteCourse }) => {
const [activeTab, setActiveTab] = useState<'lecture' | 'quiz' | 'tutor'>('lecture');
const [lectureContent, setLectureContent] = useState('');
const [loading, setLoading] = useState(false);
// Quiz State
const [quizQuestions, setQuizQuestions] = useState<QuizQuestion[]>([]);
const [quizAnswers, setQuizAnswers] = useState<number[]>([]);
const [quizSubmitted, setQuizSubmitted] = useState(false);
// Chat State
const [chatInput, setChatInput] = useState('');
const [chatHistory, setChatHistory] = useState<{sender: string, text: string}[]>([]);
const messagesEndRef = useRef<HTMLDivElement>(null);
// Derive current topic from syllabus (looping if weeks > syllabus length for demo)
const topicIndex = (week - 1) % course.syllabus.length;
const currentTopic = course.syllabus[topicIndex] || "Advanced Topics";
useEffect(() => {
loadLecture();
}, [course.id, week]);
const loadLecture = async () => {
setLoading(true);
const content = await generateLecture(course.title, currentTopic, week);
setLectureContent(content);
setLoading(false);
};
const handleStartQuiz = async () => {
setActiveTab('quiz');
if (quizQuestions.length === 0) {
setLoading(true);
const questions = await generateCourseQuiz(course.title, currentTopic);
setQuizQuestions(questions);
setQuizAnswers(new Array(questions.length).fill(-1));
setLoading(false);
}
};
const submitQuiz = () => {
let score = 0;
quizQuestions.forEach((q, i) => {
if (quizAnswers[i] === q.correctOptionIndex) score++;
});
// Grade logic: simple percentage
const finalGrade = Math.round((score / quizQuestions.length) * 100);
setQuizSubmitted(true);
// If passed, complete the course module
if (finalGrade >= 50) {
// In a real app we'd track weekly progress, here we just complete the course for the demo effect
setTimeout(() => {
if(window.confirm(`Module complete! Grade: ${finalGrade}%. Return to dashboard?`)) {
onCompleteCourse(course.id, finalGrade);
}
}, 1000);
} else {
alert(`Grade: ${finalGrade}%. You must retake the quiz.`);
setQuizSubmitted(false);
setQuizAnswers(new Array(quizQuestions.length).fill(-1));
}
};
const handleChatSend = async () => {
if(!chatInput.trim()) return;
const userMsg = chatInput;
setChatInput('');
setChatHistory(prev => [...prev, { sender: 'You', text: userMsg }]);
// Convert history for API
const apiHistory = chatHistory.map(m => ({
role: m.sender === 'You' ? 'user' : 'model',
parts: [{ text: m.text }]
}));
const response = await chatWithAdvisor(apiHistory, userMsg);
setChatHistory(prev => [...prev, { sender: 'Tutor', text: response }]);
};
return (
<div className="flex flex-col h-full bg-slate-50 animate-in fade-in slide-in-from-bottom-4 duration-500">
{/* Header */}
<div className="bg-white border-b border-slate-200 px-6 py-4 flex items-center justify-between sticky top-0 z-10 shadow-sm">
<div className="flex items-center space-x-4">
<button onClick={onBack} className="p-2 hover:bg-slate-100 rounded-full transition-colors">
<ArrowLeft className="w-5 h-5 text-slate-600" />
</button>
<div>
<h2 className="text-xl font-bold text-slate-800">{course.title}</h2>
<p className="text-sm text-slate-500">Week {week}: {currentTopic}</p>
</div>
</div>
<div className="flex space-x-1 bg-slate-100 p-1 rounded-lg">
<button
onClick={() => setActiveTab('lecture')}
className={`px-4 py-2 rounded-md text-sm font-medium transition-all ${activeTab === 'lecture' ? 'bg-white text-blue-900 shadow-sm' : 'text-slate-500 hover:text-slate-700'}`}
>
Lecture
</button>
<button
onClick={handleStartQuiz}
className={`px-4 py-2 rounded-md text-sm font-medium transition-all ${activeTab === 'quiz' ? 'bg-white text-blue-900 shadow-sm' : 'text-slate-500 hover:text-slate-700'}`}
>
Quiz
</button>
<button
onClick={() => setActiveTab('tutor')}
className={`px-4 py-2 rounded-md text-sm font-medium transition-all ${activeTab === 'tutor' ? 'bg-white text-blue-900 shadow-sm' : 'text-slate-500 hover:text-slate-700'}`}
>
AI Tutor
</button>
</div>
</div>
{/* Main Content */}
<div className="flex-1 overflow-y-auto p-6 md:p-12 max-w-5xl mx-auto w-full">
{loading ? (
<div className="flex flex-col items-center justify-center h-64 text-slate-400">
<Loader2 className="w-8 h-8 animate-spin mb-4" />
<p>Professor is preparing materials...</p>
</div>
) : (
<>
{activeTab === 'lecture' && (
<article className="prose prose-slate prose-lg max-w-none bg-white p-8 md:p-12 rounded-xl shadow-sm border border-slate-100">
<div className="flex items-center space-x-3 mb-6 pb-6 border-b border-slate-100">
<div className="bg-blue-100 p-3 rounded-lg">
<BookOpen className="w-6 h-6 text-blue-900" />
</div>
<div>
<h1 className="text-2xl font-serif font-bold text-slate-900 m-0">Lecture Notes</h1>
<span className="text-slate-500 text-sm">Authored by Virtuverse Faculty AI</span>
</div>
</div>
<ReactMarkdown>{lectureContent}</ReactMarkdown>
<div className="mt-12 flex justify-end">
<button onClick={handleStartQuiz} className="bg-blue-900 text-white px-6 py-3 rounded-lg font-bold hover:bg-blue-800 transition-colors flex items-center">
Proceed to Assessment <ArrowLeft className="w-4 h-4 ml-2 rotate-180" />
</button>
</div>
</article>
)}
{activeTab === 'quiz' && (
<div className="max-w-2xl mx-auto space-y-8">
<div className="bg-white p-8 rounded-xl shadow-sm border border-slate-200">
<h2 className="text-2xl font-serif font-bold text-slate-900 mb-6">Weekly Assessment</h2>
{quizQuestions.map((q, i) => (
<div key={i} className="mb-8 last:mb-0">
<p className="font-medium text-lg text-slate-800 mb-3">{i + 1}. {q.question}</p>
<div className="space-y-2">
{q.options.map((opt, optIndex) => (
<button
key={optIndex}
disabled={quizSubmitted}
onClick={() => {
const newAns = [...quizAnswers];
newAns[i] = optIndex;
setQuizAnswers(newAns);
}}
className={`w-full text-left p-4 rounded-lg border transition-all ${
quizSubmitted
? (optIndex === q.correctOptionIndex
? 'bg-green-100 border-green-500 text-green-800'
: (quizAnswers[i] === optIndex ? 'bg-red-100 border-red-500 text-red-800' : 'bg-slate-50 border-slate-200 text-slate-400'))
: (quizAnswers[i] === optIndex
? 'bg-blue-50 border-blue-500 text-blue-900 ring-1 ring-blue-500'
: 'bg-white border-slate-200 hover:border-blue-300 text-slate-700')
}`}
>
{opt}
</button>
))}
</div>
</div>
))}
{!quizSubmitted && (
<button
onClick={submitQuiz}
disabled={quizAnswers.includes(-1)}
className="w-full mt-6 bg-yellow-500 hover:bg-yellow-600 text-black font-bold py-3 rounded-lg disabled:opacity-50 disabled:cursor-not-allowed"
>
Submit Quiz
</button>
)}
</div>
</div>
)}
{activeTab === 'tutor' && (
<div className="max-w-2xl mx-auto h-[600px] flex flex-col bg-white rounded-xl shadow-sm border border-slate-200">
<div className="p-4 border-b border-slate-100 bg-slate-50 rounded-t-xl flex items-center space-x-3">
<div className="p-2 bg-purple-100 rounded-full">
<GraduationCap className="w-5 h-5 text-purple-700" />
</div>
<div>
<h3 className="font-bold text-slate-800">Professor AI</h3>
<p className="text-xs text-slate-500">Ask me anything about the lecture.</p>
</div>
</div>
<div className="flex-1 overflow-y-auto p-4 space-y-4 bg-slate-50/50">
{chatHistory.length === 0 && (
<div className="text-center text-slate-400 mt-12">
<MessageSquare className="w-12 h-12 mx-auto mb-2 opacity-20" />
<p>Start a conversation to clarify doubts.</p>
</div>
)}
{chatHistory.map((msg, i) => (
<div key={i} className={`flex ${msg.sender === 'You' ? 'justify-end' : 'justify-start'}`}>
<div className={`max-w-[80%] p-3 rounded-2xl ${
msg.sender === 'You'
? 'bg-blue-900 text-white rounded-br-none'
: 'bg-white border border-slate-200 text-slate-700 rounded-bl-none shadow-sm'
}`}>
<p className="text-sm">{msg.text}</p>
</div>
</div>
))}
<div ref={messagesEndRef} />
</div>
<div className="p-4 border-t border-slate-100">
<div className="flex space-x-2">
<input
type="text"
value={chatInput}
onChange={(e) => setChatInput(e.target.value)}
onKeyDown={(e) => e.key === 'Enter' && handleChatSend()}
placeholder="Type your question..."
className="flex-1 border border-slate-300 rounded-lg px-4 py-2 focus:ring-2 focus:ring-blue-900 focus:outline-none"
/>
<button
onClick={handleChatSend}
className="bg-blue-900 text-white p-2 rounded-lg hover:bg-blue-800"
>
<Send className="w-5 h-5" />
</button>
</div>
</div>
</div>
)}
</>
)}
</div>
</div>
);
};
export default Classroom;
