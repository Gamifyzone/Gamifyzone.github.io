import React, { useState } from 'react';
import { BookOpen, CheckCircle, GraduationCap, ArrowRight, Loader2 } from 'lucide-react';
import { generateEntranceExam, generateCurriculum } from '../services/geminiService';
import { QuizQuestion, Curriculum, StudentProfile } from '../types';
interface AdmissionsProps {
onComplete: (profile: StudentProfile, curriculum: Curriculum) => void;
}
const Admissions: React.FC<AdmissionsProps> = ({ onComplete }) => {
const [step, setStep] = useState<'form' | 'exam' | 'processing'>('form');
const [name, setName] = useState('');
const [field, setField] = useState('');
const [loading, setLoading] = useState(false);
const [examQuestions, setExamQuestions] = useState<QuizQuestion[]>([]);
const [answers, setAnswers] = useState<number[]>([]);
const handleStartExam = async () => {
if (!name || !field) return;
setLoading(true);
const questions = await generateEntranceExam(field);
setExamQuestions(questions);
setAnswers(new Array(questions.length).fill(-1));
setStep('exam');
setLoading(false);
};
const handleSubmitExam = async () => {
setStep('processing');
setLoading(true);
// Calculate score locally for simulation speed, though a real app might verify server-side
let score = 0;
examQuestions.forEach((q, i) => {
if (answers[i] === q.correctOptionIndex) score++;
});
// Assume 60% pass rate
const passed = (score / examQuestions.length) >= 0.6;
if (passed) {
const curriculum = await generateCurriculum(field);
const profile: StudentProfile = {
name,
fieldOfStudy: field,
gpa: 4.0, // Start fresh
creditsEarned: 0,
currentSemester: 1,
currentWeek: 1,
status: 'Enrolled',
feedbackHistory: [`Admitted to program based on entrance exam score: ${score}/${examQuestions.length}`]
};
onComplete(profile, curriculum);
} else {
alert(`Unfortunately, your score of ${score}/${examQuestions.length} was insufficient for direct admission. Please restart to try again.`);
setStep('form');
setLoading(false);
}
};
if (loading) {
return (
<div className="flex flex-col items-center justify-center h-full p-12 text-center space-y-4 animate-in fade-in duration-500">
<Loader2 className="w-16 h-16 text-blue-900 animate-spin" />
<h2 className="text-2xl font-serif text-slate-800">Processing Application...</h2>
<p className="text-slate-500">The Admissions Committee is reviewing your data.</p>
</div>
);
}
if (step === 'exam') {
return (
<div className="max-w-3xl mx-auto p-6 space-y-8 animate-in slide-in-from-right duration-500">
<div className="bg-white p-8 rounded-xl shadow-lg border-t-4 border-blue-900">
<div className="flex justify-between items-center mb-6">
<h2 className="text-2xl font-serif font-bold text-slate-900">Entrance Assessment</h2>
<span className="text-sm font-semibold text-blue-600 bg-blue-50 px-3 py-1 rounded-full">
Target: {field}
</span>
</div>
<p className="mb-6 text-slate-600">
Please answer the following questions to demonstrate your aptitude for this degree program.
</p>
<div className="space-y-8">
{examQuestions.map((q, qIndex) => (
<div key={qIndex} className="space-y-3">
<p className="font-medium text-lg text-slate-800">{qIndex + 1}. {q.question}</p>
<div className="grid grid-cols-1 gap-2">
{q.options.map((opt, oIndex) => (
<button
key={oIndex}
onClick={() => {
const newAnswers = [...answers];
newAnswers[qIndex] = oIndex;
setAnswers(newAnswers);
}}
className={`text-left px-4 py-3 rounded-lg border transition-all ${
answers[qIndex] === oIndex
? 'bg-blue-900 text-white border-blue-900'
: 'bg-white text-slate-700 border-slate-200 hover:border-blue-400'
}`}
>
{opt}
</button>
))}
</div>
</div>
))}
</div>
<div className="mt-8 flex justify-end">
<button
onClick={handleSubmitExam}
disabled={answers.includes(-1)}
className="flex items-center space-x-2 bg-yellow-500 hover:bg-yellow-600 text-black font-bold py-3 px-8 rounded-lg disabled:opacity-50 disabled:cursor-not-allowed transition-colors"
>
<span>Submit Assessment</span>
<CheckCircle className="w-5 h-5" />
</button>
</div>
</div>
</div>
);
}
return (
<div className="max-w-4xl mx-auto grid grid-cols-1 md:grid-cols-2 gap-12 items-center p-6 animate-in fade-in duration-700">
<div className="space-y-6">
<div className="inline-block p-3 bg-blue-100 rounded-2xl mb-4">
<GraduationCap className="w-8 h-8 text-blue-900" />
</div>
<h1 className="text-4xl md:text-5xl font-serif font-bold text-slate-900 leading-tight">
Begin Your <br/><span className="text-blue-900">Academic Journey</span>
</h1>
<p className="text-lg text-slate-600">
Virtuverse University offers a rigorous, AI-driven 1-year degree program tailored to your goals.
Enroll today to start your personalized curriculum.
</p>
<div className="space-y-4 pt-4">
<div className="flex items-center space-x-3 text-slate-700">
<CheckCircle className="w-5 h-5 text-green-500" />
<span>Personalized AI Curriculum</span>
</div>
<div className="flex items-center space-x-3 text-slate-700">
<CheckCircle className="w-5 h-5 text-green-500" />
<span>Real-time Interactive Tutoring</span>
</div>
<div className="flex items-center space-x-3 text-slate-700">
<CheckCircle className="w-5 h-5 text-green-500" />
<span>Industry-Standard Certification</span>
</div>
</div>
</div>
<div className="bg-white p-8 rounded-2xl shadow-xl border border-slate-100">
<h3 className="text-xl font-bold text-slate-800 mb-6">Application Form</h3>
<div className="space-y-4">
<div>
<label className="block text-sm font-medium text-slate-700 mb-1">Full Name</label>
<input
type="text"
value={name}
onChange={(e) => setName(e.target.value)}
className="w-full px-4 py-2 rounded-lg border border-slate-300 focus:ring-2 focus:ring-blue-900 focus:outline-none"
placeholder="e.g. Alex Morgan"
/>
</div>
<div>
<label className="block text-sm font-medium text-slate-700 mb-1">Intended Degree</label>
<input
type="text"
value={field}
onChange={(e) => setField(e.target.value)}
className="w-full px-4 py-2 rounded-lg border border-slate-300 focus:ring-2 focus:ring-blue-900 focus:outline-none"
placeholder="e.g. Computer Science, Digital Marketing..."
/>
<p className="text-xs text-slate-500 mt-1">Our AI will generate a syllabus based on this field.</p>
</div>
<button
onClick={handleStartExam}
disabled={!name || !field}
className="w-full mt-4 bg-blue-900 hover:bg-blue-800 text-white font-semibold py-3 rounded-lg flex items-center justify-center space-x-2 disabled:opacity-50 transition-colors"
>
<span>Start Entrance Exam</span>
<ArrowRight className="w-4 h-4" />
</button>
</div>
</div>
</div>
);
};
export default Admissions;
