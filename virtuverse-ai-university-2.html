import { GoogleGenAI, Type } from "@google/genai";
import { Curriculum, QuizQuestion, ExamResult } from '../types';
// Initialize Gemini Client
// @ts-ignore - process is global
const apiKey = process.env.API_KEY || '';
const ai = new GoogleGenAI({ apiKey });
const MODEL_FAST = 'gemini-3-flash-preview';
const MODEL_REASONING = 'gemini-3-pro-preview';
/**
*GENERATE ENTRANCE EXAM
*/
export const generateEntranceExam = async (fieldOfStudy: string): Promise<QuizQuestion[]> => {
try {
const response = await ai.models.generateContent({
model: MODEL_FAST,
contents: `Generate a diagnostic entrance exam for a student applying for a degree in ${fieldOfStudy}.
Create 5 multiple-choice questions ranging from basic to intermediate difficulty to test their aptitude.`,
config: {
responseMimeType: "application/json",
responseSchema: {
type: Type.ARRAY,
items: {
type: Type.OBJECT,
properties: {
question: { type: Type.STRING },
options: {
type: Type.ARRAY,
items: { type: Type.STRING }
},
correctOptionIndex: { type: Type.INTEGER, description: "Index of the correct answer (0-3)" }
},
required: ["question", "options", "correctOptionIndex"]
}
}
}
});
if (response.text) {
return JSON.parse(response.text) as QuizQuestion[];
}
throw new Error("No data returned");
} catch (error) {
console.error("Failed to generate exam:", error);
return [];
}
};
/**
*GENERATE CURRICULUM
*/
export const generateCurriculum = async (fieldOfStudy: string): Promise<Curriculum> => {
try {
const response = await ai.models.generateContent({
model: MODEL_FAST,
contents: `Create a comprehensive 1-year (2 semester) curriculum for a degree in ${fieldOfStudy}.
The curriculum should have a professional title and 6 core courses.
Each course must have a list of 4 key weekly topics (syllabus) that covers the 6-month semester (simplified for simulation).`,
config: {
responseMimeType: "application/json",
responseSchema: {
type: Type.OBJECT,
properties: {
degreeTitle: { type: Type.STRING },
courses: {
type: Type.ARRAY,
items: {
type: Type.OBJECT,
properties: {
id: { type: Type.STRING },
code: { type: Type.STRING },
title: { type: Type.STRING },
description: { type: Type.STRING },
credits: { type: Type.INTEGER },
syllabus: {
type: Type.ARRAY,
items: { type: Type.STRING },
description: "List of 4 major topics/modules for this course"
}
},
required: ["id", "code", "title", "description", "credits", "syllabus"]
}
}
},
required: ["degreeTitle", "courses"]
}
}
});
if (response.text) {
const data = JSON.parse(response.text);
// Hydrate with defaults
data.courses = data.courses.map((c: any) => ({ ...c, completed: false }));
return data as Curriculum;
}
throw new Error("Failed to generate curriculum");
} catch (error) {
console.error(error);
throw error;
}
};
/**
*GENERATE LECTURE CONTENT
*/
export const generateLecture = async (courseTitle: string, topic: string, week: number): Promise<string> => {
try {
const response = await ai.models.generateContent({
model: MODEL_REASONING,
contents: `You are a Professor at Virtuverse University.
Teach a lecture for the course "${courseTitle}" on the topic: "${topic}".
This is Week ${week} of the semester.
Structure:
1.Introduction & Learning Goals
2.Core Concept (Deep Dive) with Analogies
3.Real-world Application / Case Study
4.Summary
Tone: Academic, authoritative, yet engaging. Use Markdown formatting. Keep it concise (approx 400 words).`,
});
return response.text || "Lecture generation failed.";
} catch (error) {
console.error(error);
return "Professor is currently unavailable. Please try again later.";
}
};
/**
*GENERATE COURSE QUIZ
*/
export const generateCourseQuiz = async (courseTitle: string, topic: string): Promise<QuizQuestion[]> => {
try {
const response = await ai.models.generateContent({
model: MODEL_FAST,
contents: `Generate a short quiz (3 questions) to test understanding of "${topic}" in the course "${courseTitle}".`,
config: {
responseMimeType: "application/json",
responseSchema: {
type: Type.ARRAY,
items: {
type: Type.OBJECT,
properties: {
question: { type: Type.STRING },
options: { type: Type.ARRAY, items: { type: Type.STRING } },
correctOptionIndex: { type: Type.INTEGER }
},
required: ["question", "options", "correctOptionIndex"]
}
}
}
});
return response.text ? JSON.parse(response.text) : [];
} catch (error) {
console.error(error);
return [];
}
};
/**
*ADVISOR CHAT
*/
export const chatWithAdvisor = async (history: {role: string, parts: {text: string}[]}[], message: string): Promise<string> => {
try {
const chat = ai.chats.create({
model: MODEL_FAST,
config: {
systemInstruction: "You are the Academic Advisor at Virtuverse University. Your goal is to guide the student, offer career advice, and ensure they stay on track. Be professional, supportive, but strict about academic integrity."
},
history: history
});
const result = await chat.sendMessage({ message });
return result.text || "I apologize, I cannot answer that right now.";
} catch (error) {
console.error(error);
return "System error in Advisor module.";
}
};
